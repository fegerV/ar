    """Verify backup integrity without restoring."""
    metadata_path = backup_path.with_suffix(".json")
    if not metadata_path.exists():
        return {"success": False, "error": "Metadata file not found"}
    
    with open(metadata_path) as f:
        metadata = json.load(f)
    
    # Verify checksum
    current_checksum = self._calculate_checksum(backup_path)
    if current_checksum != metadata.get("checksum"):
        return {"success": False, "error": "Checksum mismatch - backup may be corrupted"}
    
    # For tar files, verify structure
    if backup_path.suffix in ['.tar', '.tgz'] or '.tar.' in backup_path.name:
        try:
            with tarfile.open(backup_path, "r:*") as tar:
                tar.getmembers()  # Verify can read all members
        except Exception as e:
            return {"success": False, "error": f"Archive corrupted: {str(e)}"}
    
    return {"success": True, "message": "Backup is valid"}
