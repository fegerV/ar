```python
@router.get("/history")
async def get_metrics_history(
    hours: int = 24,
    metric_type: Optional[str] = None,
    db: Session = Depends(get_db)
):
    """Get historical metrics data."""
    cutoff = datetime.utcnow() - timedelta(hours=hours)
    
    query = db.query(SystemMetric).filter(
        SystemMetric.timestamp >= cutoff
    )
    
    if metric_type:
        query = query.filter(SystemMetric.metric_type == metric_type)
    
    metrics = query.order_by(SystemMetric.timestamp).all()
    
    return {
        "timestamps": [m.timestamp.isoformat() for m in metrics],
        "values": [m.value for m in metrics],
        "metadata": [m.metadata for m in metrics]
    }
```
